My BoilerPlate

Site Web responsive, prêt à l'emploi, développé avec Python/Django.

Explication du projet :

MyBoilerplate est un projet de démarrage rapide pour la création de sites web structurés. Il intègre une zone de connexion sécurisée, une barre de navigation, un pied de page, ainsi qu'une gestion modulaire de l'interface via des fragments. Idéal pour poser rapidement les fondations d'une application web moderne.

from django.contrib.auth import authenticate, login, logout
from django.contrib import messages
from django.core.cache import cache
from django.shortcuts import render, redirect
from django.contrib.auth.decorators import login_required
from django.utils import timezone
from django.views.decorators.cache import cache_control


@cache_control(no_cache=True, must_revalidate=True, no_store=True)
def login_view(request):
    if request.method == "POST":
        username = request.POST['username']
        password = request.POST['password']
        user = authenticate(request, username=username, password=password)

        if user is not None:
            login(request, user)
            messages.success(request, "Connexion réussie.")
            return redirect("index")
        else:
            messages.error(request, "Identifiant ou mot de passe "
                                    "incorrect.")

    return render(request, "myboilerplate/login.html",
                  {'hide_footer_nav': True})


@login_required
def logout_view(request):
    logout(request)
    if not request.user.is_authenticated:
        print("Utilisateur déconnecté avec succès")
        messages.success(request, "Déconnexion réussie.")
    else:
        print("Échec de la déconnexion")
    return redirect('login')


@cache_control(no_cache=True, must_revalidate=True, no_store=True)
@login_required
def index(request):
      # Récupération du nom d'utilisateur de la personne connectée ou "Anonymous" si non authentifié
    username = request.user.username if request.user.is_authenticated else 'Anonymous'
# Calcul de la date d'aujourd'hui
    today = timezone.localdate()

    return render(request, "myboilerplate/index.html", {
        'today': today,
        'username': username,
            })
