Spotnix

Serveur multimédia local en Python/Django.

Explication du projet

Spotnix est un serveur multimédia personnel conçu pour centraliser et diffuser localement une bibliothèque de films et de séries, offrant une expérience de navigation immersive similaire aux grandes plateformes de streaming. Développé en Python/Django, ce projet va au-delà d'un simple lecteur : il agit comme un agrégateur intelligent de métadonnées, et permet de télécharger le film choisit au format *.iso. L'application interroge en temps réel l'API TMDb pour enrichir automatiquement chaque titre avec ses jaquettes, synopsis, genres et années de sortie, transformant un simple répertoire de fichiers en une interface visuelle riche et organisée. J'ai conçu une architecture modulaire permettant une authentification sécurisée des utilisateurs, le tout dans une interface responsive et esthétique. Défis techniques : Intégration API : Gestion asynchrone des requêtes vers TMDb pour récupérer et mettre en cache les métadonnées sans ralentir l'interface. Expérience Utilisateur : Création d'une interface "Netflix-like" avec un design sombre, des transitions fluides et une adaptation parfaite aux écrans mobiles. L'ajout d'un nouveau film dans le dossier est automatiquement ajouté à la liste des films disponibles. Architecture : Conception d'un backend Django robuste capable de scanner, indexer et servir des fichiers multimédias locaux via le réseau. Résultat : Une solution autonome hébergée sur Raspberry Pi 5, permettant de transformer n'importe quel ordinateur ou NAS en un centre de divertissement personnel, sécurisé et entièrement personnalisable, sans dépendre des services cloud tiers

#@cache_control(no_cache=True, must_revalidate=True, no_store=True)
@login_required
def affichage_des_videos(request):
    """
    Affiche la liste détaillée des vidéos présentes dans le dossier défini.
    Utilise un cache JSON pour éviter les appels répétitifs à TMDb.
    """

    # 1. Charger le cache des métadonnées
    films = charger_cache() or {}
    
    # 2. Définir le chemin vers le dossier contenant les vidéos
    # chemin relatif: # dossier_vers_videos = os.path.join(settings.BASE_DIR, 'medias_data/videos')
    ou_sont_mes_videos = settings.CHEMIN_VIDEOS
    extensions_des_videos = ('.mp4', '.mkv', '.avi', '.mov', '.wmv', '.flv', '.webm', '.mpeg', '.iso', '.m2ts')
    
    # Initialiser une liste vide pour stocker les objets vidéo
    liste_des_videos_disponibles = []
    
    # Vérifier que le dossier existe
    if os.path.exists(ou_sont_mes_videos):
        # Parcourir chaque élément du dossier
        for fichier in os.listdir(ou_sont_mes_videos):
            # Ne garder que les fichiers MKV (insensible à la casse)
            if fichier.lower().endswith(extensions_des_videos):
                chemin_complet = os.path.join(ou_sont_mes_videos, fichier)
                titre_film = os.path.splitext(fichier)[0]
            
                # 3. Récupérer les métadonnées depuis le cache ou TMDb
                metadonnees = films.get(titre_film)
                
                if not metadonnees:
                    metadonnees = recuperer_metadonnees_tmdb(titre_film)
                    print(f"\033[91mLes métadonnées ont été récupérées depuis TMDB !\033[0m")
                    if metadonnees:
                         films[titre_film] = metadonnees
                    else:
                        print(f"\033[91mTitre du film manquant : \033[0m {titre_film}")
                        continue  # ne pas ajouter ce film
                
                liste_des_videos_disponibles.append({
                    'nom': titre_film,
                    'chemin_complet': chemin_complet,
                    'extension': os.path.splitext(fichier)[1].lower(),
                    'id_tmdb': metadonnees['id_tmdb'],
                    'affiche': metadonnees['affiche'],
                    'metadonnees': metadonnees
                })
                    
    # 4. Sauvegarder le cache mis à jour
    sauvegarder_cache(films) 
    print(f"\033[92mSauvegarde du cache !\033[0m")            

    # 5. Préparer les données pour le template
    contexte = {
        'videos': liste_des_videos_disponibles,
        'films' : films
    }

    # 6. Rendre le template
    return render(request, "medias/videos.html", contexte)

def refresh_cache(request):
    films = affichage_des_videos()
    sauvegarder_cache(films)
    return JsonResponse({"status": "ok", "message": "Cache mis à jour"})

def recuperer_metadonnees_tmdb(titre_film):
    # clé API TMDb
    tmdb.API_KEY = settings.TMDB_API_KEY
    """
    Recherche un film par titre sur TMDb et retourne ses métadonnées principales.
    
    Args:
        titre_film (str): titre du film à rechercher.
    
    Returns:
        dict ou None: Dictionnaire avec titre, année, affiche, synopsis, ou None si non trouvé.
    """
    recherche = tmdb.Search()
    reponse = recherche.movie(query=titre_film, language='fr-FR')
    results = reponse.get('results', [])

    if not results:
        return None

    film = results[0]  # prendre le premier résultat

    # Récupérer les détails du film
    movie = tmdb.Movies(film['id'])
    details = movie.info(language='fr-FR')  # détails en français
    genres = [genre['name'] for genre in details.get('genres', [])]  # liste des genres
    duree = details.get('runtime')  # durée en minute

    return {
        'titre': details.get('title'),
        'affiche': f"https://image.tmdb.org/t/p/w500{details.get('poster_path')}" if details.get('poster_path') else None,
        'id_tmdb': details.get('id')  # ID TMDb pour requêtes future
    }

def detail_film(request, id_tmdb):
    import tmdbsimple as tmdb
    from django.conf import settings
    from django.http import HttpResponse

    # 1. On récupère le chemin réel envoyé dans l'URL
    chemin_video = request.GET.get('chemin')  
    if not chemin_video:
        return HttpResponse("Chemin vidéo manquant.", status=400)

    # 2. On fabrique une URL publique à partir du nom de fichier
    nom_fichier = os.path.basename(chemin_video)
    chemin_video_url = f"{settings.MEDIA_URL}{nom_fichier}"
    #2.1 On récupère l'extension du fichier vidéo
    _, extension = os.path.splitext(nom_fichier)
    extension = extension.upper().lstrip('.')

    # 3. Définir la clé API TMDb
    tmdb.API_KEY = settings.TMDB_API_KEY

    # 4. On récupère les infos TMDb
    movie = tmdb.Movies(id_tmdb)
    details = movie.info(language='fr-FR')
    genres = [genre['name'] for genre in details.get('genres', [])]
    duree = details.get('runtime')
    taille_mo = round(os.path.getsize(chemin_video) / (1024 * 1024), 2)
    synopsis = details.get('overview')
    
    # 5. Construire l’URL complète vidéo (avec domaine et port)
    domaine = request.get_host()  # ex: '192.168.1.98:8000'
    url_video_complete = f"http://{domaine}{chemin_video_url}"

    # 6. Préparer le contexte pour le template
    contexte = {
        'titre': details.get('title'),
        'annee': details.get('release_date', '')[:4],
        'affiche': f"https://image.tmdb.org/t/p/w500{details.get('poster_path')}" if details.get('poster_path') else None,
        'backdrop': f"https://image.tmdb.org/t/p/w1280{details.get('backdrop_path')}" if details.get('backdrop_path') else None,
        'synopsis': synopsis,
        'genres': genres,
        'duree': duree,
        'taille_mo': taille_mo,
        'chemin_video': chemin_video_url,
        'url_video_complete': url_video_complete,
        'extension': extension,
    }
    return render(request, 'medias/detail_film.html', contexte)
