CALL-ASSIGN

Application de gestion de transferts d'appels.

Explication du projet

Call-Assign est une application web d'administration critique développée pour Linkso, garantissant la continuité des services d'astreinte téléphonique 24h/24 et 7j/7 du cabinet médical Branchet. Avant ce projet, la gestion des affectations d'urgence (pôles médico-légal, infectiologie, cybersécurité, psychologie) reposait sur des requêtes SQL manuelles, une méthode lente, risquée et inaccessible aux équipes non-techniques. J'ai conçu et déployé cette solution Django/PostgreSQL pour remplacer ce processus obsolète par une interface intuitive et sécurisée. L'application permet désormais aux administrateurs de gérer dynamiquement les transferts d'appels vers Asterisk, de définir des plannings d'astreinte complexes avec gestion des conflits horaires, et d'attribuer des priorités critiques en temps réel. Défis techniques relevés : Intégration système : Conception d'une logique métier robuste synchronisant la base de données avec le PBX Asterisk pour un routage d'appels instantané. Sécurité & Validation : Implémentation d'une validation stricte des numéros de téléphone (via intl-tel-input et phonenumbers) et gestion fine des droits utilisateurs pour empêcher les modifications non autorisées. Expérience utilisateur : Développement d'un tableau de bord interactif avec DataTables pour filtrer, trier et visualiser des centaines d'affectations, avec un système de notifications en temps réel pour les actions critiques (activation/désactivation). Impact : Cette solution a réduit le temps de mise à jour des plannings de plusieurs heures à quelques minutes, éliminé les erreurs de saisie humaine et garanti une disponibilité téléphonique ininterrompue pour les professionnels de santé en situation d'urgence.

@method_decorator(login_required, name='dispatch')
class Assigns(View):

    def dispatch(self, request, *args, **kwargs):
        return super().dispatch(request, *args, **kwargs)

    def get(self, request):
        """
        AFFICHAGE DES AFFECTATIONS
        Permet l'affichage de la page de Gestion des Affectations, d'en modifier une,
        de la désactiver, ou de la réactiver.
        """

        files_transferables = QueueTransfert.objects.filter(isactive='Y')
        transfert_map = QueueTransfertMap.objects.all()

        # Instancier le formulaire avec les transferts et les personnels
        form_queue_transfert_map = QueueTransfertMapForm(
            request.POST,
            transferts_actifs=files_transferables,
            # Récupérer les titres, noms et prénoms du personnel affecté
            personnel_actif=QueueTransfertUser.objects.filter(isactive='Y')
        )

        return render(
            request, 'htwentyfour/affectations.html',
            {
                'transferts_actifs': files_transferables,
                'affectation': transfert_map,
                'form_queue_transfert_map': form_queue_transfert_map,
            })

    def post(self, request):
        """
        CREATION DES AFFECTATIONS
        """
        # On instancie un object QueueTransfertMapForm à partir de request.POST
        form_queue_transfert_map = QueueTransfertMapForm(request.POST)

        try:
            recup_id(QueueTransfert, request.POST.get('queue_transfert'))
            recup_id(QueueTransfertUser, request.POST.get('queue_transfert_user'))
            new_affectation = form_queue_transfert_map.save(commit=False)

            error = check_prio(new_affectation, request.POST.get('queue_transfert'))

            if error:
                messages.error(request, error)
                return self.get(request)

            messages.success(request, "Affectation créée !")
            new_affectation.save()
            return redirect('htwentyfour:assigns')
        except ValueError:
            for error in form_queue_transfert_map.errors['__all__']:
                messages.error(request, error)
        except QueueTransfert.DoesNotExist:
            messages.error(request, "Problème avec la récupération de l'ID de l'astreinte choisie.")
        except QueueTransfertMap.DoesNotExist:
            messages.error(request, "Problème lors de la récupération de l'ID du membre concerné.")

        return self.get(request)

@login_required
def display_modify_assigns(request, id):
    # Si la méthode est POST, appelle modify_assigns
    if request.method == 'POST':
        assignsTools.modify_assigns(request, id)
        return redirect('htwentyfour:assigns')

    
    identifiant_affectation = QueueTransfertMap.objects.get(queue_transfert_map_id=id)
    identifiant_membre = identifiant_affectation.queue_transfert_user_id  

    # Récupérer toutes les affectations et tous les membres du personnel dans la base de données
    liste_affectations = set(QueueTransfert.objects.all())  
    liste_personnel = set(QueueTransfertUser.objects.filter(isactive="Y")) 

    # Création d'une instance de formulaire `QueueTransfertMapForm`
    form_queue_transfert_map = QueueTransfertMapForm(
        instance=identifiant_affectation)  

    # Modification des choix du champ 'transferts' dans le formulaire avec les transferts uniques récupérés de la BDD
    form_queue_transfert_map.fields['queue_transfert'].choices = [
        (queue_transfert.queue_transfert_id, queue_transfert.name) for queue_transfert in liste_affectations]

    form_queue_transfert_map.fields['queue_transfert_user'].choices = [
        (queue_transfert_user.queue_transfert_user_id,
         f"{queue_transfert_user.titre} {queue_transfert_user.name} {queue_transfert_user.firstname}")
        for queue_transfert_user in liste_personnel
    ]

    identifiant_transfert = identifiant_affectation.queue_transfert_id  # print(f"VIEWS > affichage_modifier_affectation > identifiant_transfert {identifiant_transfert}")
    start_time = identifiant_affectation.start_time  # print(f"VIEWS > affichage_modifier_affectation > start_time : {start_time}")
    end_time = identifiant_affectation.end_time  # print(f"VIEWS > affichage_modifier_affectation > end_time : {end_time}")

    # Mise à jour UPDATED
    temporaire_timestamp_now = timezone.now()
    if temporaire_timestamp_now.tzinfo is None:
        # Si timestamp_now est naïf, le convertir en un objet aware avec le fuseau horaire local
        temporaire_timestamp_now = timezone.localtime(temporaire_timestamp_now)
    formatted_timestamp = temporaire_timestamp_now.strftime('%Y-%m-%d %H:%M:%S.%f')

    return render(
        request, 'htwentyfour/modifier_affectations.html',
        {
            'transferts_actifs': QueueTransfert.objects.all(),
            'personnel_actif': QueueTransfertUser.objects.all(),
            'affectation': QueueTransfertMap.objects.all(),
            'id_affectation': identifiant_affectation,
            'id_user': identifiant_membre,
            'id_transfert': identifiant_transfert,
            'start_time': start_time,
            'end_time': end_time,
            'form_queue_transfert_map': form_queue_transfert_map,
            'updated': formatted_timestamp
        })


def desactivate_assigns(request: WSGIRequest, id):
    try:
        set_isactive_assign(id, 'N')
        messages.success(request, "Désactivation de l'astreinte réussie")
    except QueueTransfertMap.DoesNotExist as e:
        messages.error(request, "Problème lors de la désactivation de l'astreinte : ")
        print(f"Erreur rencontrée : {e}")
    return redirect('htwentyfour:assigns')


def reactivate_assigns(request: WSGIRequest, id):
    # Récupérer l'affectation
    affectation = get_object_or_404(QueueTransfertMap, queue_transfert_map_id=id)
    form_queue_transfert_map = QueueTransfertMapForm(request.POST, instance=affectation)

    # Récupérer les IDs associés
    id_transfert_recupere = affectation.queue_transfert_id  
    id_affectation_recupere = affectation.queue_transfert_map_id
    id_personnel_recupere = affectation.queue_transfert_user_id

    # Récupérer le membre associé à cette astreinte
    personnel_concerne = get_object_or_404(QueueTransfertUser, queue_transfert_user_id=id_personnel_recupere)

    # Vérifier si le membre est actif
    if personnel_concerne.isactive == 'N':
        messages.error(request,
                       f"Impossible de réactiver cette astreinte si {personnel_concerne.titre} {personnel_concerne.name} {personnel_concerne.firstname} est désactivé.")
        return redirect('htwentyfour:assigns')

    try:
        error = check_prio(affectation, id_transfert_recupere)

        if error is not None:
            messages.error(request, error)
            return redirect('htwentyfour:assigns')  # Rediriger en cas d'échec

        # Si check_prio est un succès, poursuivre avec set_isactive_assign
        set_isactive_assign(id_affectation_recupere, 'Y')
        messages.success(request, "Réactivation de l'astreinte réussie")

    except QueueTransfertMap.DoesNotExist as e:
        messages.error(request, f"Problème lors de la réactivation de l'astreinte")
        print(f"Problème lors de la réactivation de l'astreinte : {e}")
    return redirect('htwentyfour:assigns')

                table.column(2).search('^' + transfert_recupere + '$', true, false).draw();
            }

            else {
                table.column(2).search('').draw();
            }
        });

document.addEventListener('DOMContentLoaded', function () {
    let commentaireField = document.getElementById('commentaire');
    let maxLength = 150;

    commentaireField.addEventListener('input', function () {
        let currentLength = commentaireField.value.length;
        if (currentLength > maxLength) {
            commentaireField.value = commentaireField.value.slice(0, maxLength);
        }
        let remainingChars = maxLength - currentLength;
        let message = remainingChars >= 0 ? `${remainingChars} caractères restants` : `Limite de ${maxLength} caractères atteinte`;
        document.getElementById('charCountMessage').innerText = message;
    });
});
